===== DOCKER COMPOSE CONFIG =====
name: github-actions-container-pipeline
services:
  app:
    build:
      context: /home/ubuntu/github-actions-container-pipeline
      dockerfile: Dockerfile
    environment:
      NODE_ENV: production
    healthcheck:
      test:
        - CMD-SHELL
        - 'node -e "require(''http'').get(''http://localhost:3000/health'', (res) => { process.exit(res.statusCode === 200 ? 0 : 1) }).on(''error'', () => process.exit(1))"'
      timeout: 5s
      interval: 10s
      retries: 5
      start_period: 20s
    image: github-actions-container-pipeline:local
    networks:
      default: null
    ports:
      - mode: ingress
        target: 3000
        published: "3000"
        protocol: tcp
    restart: unless-stopped
  nginx:
    depends_on:
      app:
        condition: service_healthy
        required: true
    healthcheck:
      test:
        - CMD-SHELL
        - wget -q --spider http://localhost/health || exit 1
      timeout: 5s
      interval: 10s
      retries: 3
    image: nginx:alpine
    networks:
      default: null
    ports:
      - mode: ingress
        target: 80
        published: "8080"
        protocol: tcp
    restart: unless-stopped
    volumes:
      - type: bind
        source: /home/ubuntu/github-actions-container-pipeline/nginx.conf
        target: /etc/nginx/nginx.conf
        read_only: true
        bind:
          create_host_path: true
networks:
  default:
    name: github-actions-container-pipeline_default
